423  // these log weights may be very negative. If that's the case, exponentiating them may cause underflow
424  // so we use the "log-exp-sum" trick
425  // actually not quite...we just shift the log-weights because after they're exponentiated
426  // they have the same normalized probabilities
427 
428  // Create the distribution with exponentiated log-weights
429  // subtract the max first to prevent underflow
430  // normalization is taken care of by std::discrete_distribution
431  std::array<float_t, N> w;
432  float_t m = *std::max_element(logWts.begin(), logWts.end());
433  std::transform(logWts.begin(), logWts.end(), w.begin(),
434  [&m](const float_t& d) -> float_t { return std::exp(d-m); } );
435  std::discrete_distribution<> kGen(w.begin(), w.end());
