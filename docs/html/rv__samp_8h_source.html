726 template<size_t N, typename float_t>
727 std::array<unsigned int, N> k_gen<N, float_t>::sample(const std::array<float_t, N> &logWts)
728 {
729  // these log weights may be very negative. If that's the case, exponentiating them may cause underflow
730  // so we use the "log-exp-sum" trick
731  // actually not quite...we just shift the log-weights because after they're exponentiated
732  // they have the same normalized probabilities
733 
734  // Create the distribution with exponentiated log-weights
735  // subtract the max first to prevent underflow
736  // normalization is taken care of by std::discrete_distribution
737  std::array<float_t, N> w;
738  float_t m = *std::max_element(logWts.begin(), logWts.end());
739  std::transform(logWts.begin(), logWts.end(), w.begin(),
740  [&m](const float_t& d) -> float_t { return std::exp(d-m); } );
741  std::discrete_distribution<> kGen(w.begin(), w.end());
742 
743  // sample and return ks
744  std::array<unsigned int, N> ks;
745  for(size_t i = 0; i < N; ++i){
746  ks[i] = kGen(this->m_rng);
747  }
748  return ks;
749 }
