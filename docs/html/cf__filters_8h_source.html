199 void kalman<dimstate,dimobs,diminput>::updatePrior(const ssMat &stateTransMat,
200  const ssMat &cholStateVar,

202  const isv &inputData)

210 template<size_t dimstate, size_t dimobs, size_t diminput>
211 void kalman<dimstate,dimobs,diminput>::updatePosterior(const osv &yt,
212  const osMat &obsMat,
213  const oiMat &obsInptAffector,
214  const isv &inputData,
215  const osMat &cholObsVar)
216 {
217  osMat R = cholObsVar.transpose() * cholObsVar; //obs
218  osMat sigma = obsMat * m_predVar * obsMat.transpose() + R; // pred or APA' + R
219  osMat symSigma = (sigma.transpose() + sigma )/2.0; // ensure symmetric
220  osMat siginv = symSigma.inverse();
221  ssMat K = m_predVar * obsMat.transpose() * siginv;
222  osv obsPred = obsMat * m_predMean + obsInptAffector * inputData;
223  osv innov = yt - obsPred;
224  m_filtMean = m_predMean + K*innov;
225  m_filtVar = m_predVar - K * obsMat * m_predVar;
226 
227  // conditional likelihood stuff
228  osMat quadForm = innov.transpose() * siginv * innov;
229  osMat cholSig ( sigma.llt().matrixL() );
230  double logDet = 2.0*cholSig.diagonal().array().log().sum();
231  m_lastLogCondLike = -.5*innov.rows()*log(2*m_pi) - .5*logDet - .5*quadForm(0,0);
232 }

395 template<size_t dimstate, size_t dimobs>
396 void hmm<dimstate,dimobs>::update(const ssv &condDensVec)
397 {
398  if (!m_fresh) // hasn't seen data before and so filtVec is just time 1 state prior
399  {
400  m_filtVec = m_filtVec.cwiseProduct( condDensVec ); // now it's p(x_1, y_1)
401  m_lastCondLike = m_filtVec.sum();
402  m_filtVec /= m_lastCondLike;
403  m_fresh = true;
404 
405  } else { // has seen data before
406  m_filtVec = m_transMatTranspose * m_filtVec; // now p(x_t |y_{1:t-1})
407  m_filtVec = m_filtVec.cwiseProduct( condDensVec ); // now p(y_t,x_t|y_{1:t-1})
408  m_lastCondLike = m_filtVec.sum();
409  m_filtVec /= m_lastCondLike; // now p(x_t|y_{1:t})
410  }
411 }
