221 void kalman<dimstate,dimobs,diminput,float_t>::updatePrior(const ssMat &stateTransMat,
222  const ssMat &cholStateVar,
223  const siMat &stateInptAffector,
224  const isv &inputData)
225 {
226  ssMat Q = cholStateVar.transpose() * cholStateVar;
227  m_predMean = stateTransMat * m_filtMean + stateInptAffector * inputData;

232 template<size_t dimstate, size_t dimobs, size_t diminput, typename float_t>
233 void kalman<dimstate,dimobs,diminput,float_t>::updatePosterior(const osv &yt,
234  const osMat &obsMat,
235  const oiMat &obsInptAffector,
236  const isv &inputData,
237  const osMat &cholObsVar)
238 {
239  osMat R = cholObsVar.transpose() * cholObsVar; //obs
240  osMat sigma = obsMat * m_predVar * obsMat.transpose() + R; // pred or APA' + R
241  osMat symSigma = (sigma.transpose() + sigma )/2.0; // ensure symmetric
242  osMat siginv = symSigma.inverse();
243  ssMat K = m_predVar * obsMat.transpose() * siginv;
244  osv obsPred = obsMat * m_predMean + obsInptAffector * inputData;
245  osv innov = yt - obsPred;
246  m_filtMean = m_predMean + K*innov;
247  m_filtVar = m_predVar - K * obsMat * m_predVar;
248 
249  // conditional likelihood stuff
250  osMat quadForm = innov.transpose() * siginv * innov;
251  osMat cholSig ( sigma.llt().matrixL() );
252  float_t logDet = 2.0*cholSig.diagonal().array().log().sum();
253  m_lastLogCondLike = -.5*innov.rows()*log(2*m_pi) - .5*logDet - .5*quadForm(0,0);
254 }

284  const osMat &obsMat,

427 template<size_t dimstate, size_t dimobs, typename float_t>
428 void hmm<dimstate,dimobs,float_t>::update(const ssv &condDensVec)
429 {
430  if (!m_fresh) // hasn't seen data before and so filtVec is just time 1 state prior
431  {
432  m_filtVec = m_filtVec.cwiseProduct( condDensVec ); // now it's p(x_1, y_1)
433  m_lastCondLike = m_filtVec.sum();
434  m_filtVec /= m_lastCondLike;
435  m_fresh = true;
436 
437  } else { // has seen data before
438  m_filtVec = m_transMatTranspose * m_filtVec; // now p(x_t |y_{1:t-1})
439  m_filtVec = m_filtVec.cwiseProduct( condDensVec ); // now p(y_t,x_t|y_{1:t-1})
440  m_lastCondLike = m_filtVec.sum();
441  m_filtVec /= m_lastCondLike; // now p(x_t|y_{1:t})
442  }
443 }
