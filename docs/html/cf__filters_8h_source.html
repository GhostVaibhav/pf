181 void kalman<dimstate,dimobs,diminput>::updatePrior(const ssMat &stateTransMat,

192 template<size_t dimstate, size_t dimobs, size_t diminput>
193 void kalman<dimstate,dimobs,diminput>::updatePosterior(const osv &yt,
194  const osMat &obsMat,
195  const oiMat &obsInptAffector,
196  const isv &inputData,
197  const osMat &cholObsVar)
198 {
199  osMat R = cholObsVar.transpose() * cholObsVar; //obs
200  osMat sigma = obsMat * m_predVar * obsMat.transpose() + R; // pred or APA' + R
201  osMat symSigma = (sigma.transpose() + sigma )/2.0; // ensure symmetric
202  osMat siginv = symSigma.inverse();
203  ssMat K = m_predVar * obsMat.transpose() * siginv;
204  osVec obsPred = obsMat * m_predMean + obsInptAffector * inputData;
205  osv innov = yt - obsPred;
206  m_filtMean = m_predMean + K*innov;
207  m_filtVar = m_predVar - K * obsMat * m_predVar;
208 
209  // conditional likelihood stuff
210  osMat quadForm = innov.transpose() * siginv * innov;
211  osMat cholSig ( sigma.llt().matrixL() );
212  double logDet = 2.0*cholSig.diagonal().array().log().sum();
213  m_lastLogCondLike = -.5*innov.rows()*log(2*pi) - .5*logDet - .5*quadForm(0,0);
214 }

243  const isv &inData,
244  const osMat &obsMat,

352 template<size_t dimstate, size_t dimobs>
353 void hmm<dimstate,dimobs>::::update(const osv &yt, const ssv &condDensVec)
354 {
355  if (!m_fresh) // hasn't seen data before and so filtVec is just time 1 state prior
356  {
357  m_filtVec = m_filtVec.cwiseProduct( condDensVec ); // now it's p(x_1, y_1)
358  m_lastCondLike = m_filtVec.sum();
359  m_filtVec /= m_lastCondLike;
360  m_fresh = true;
361 
362  } else { // has seen data before
363  m_filtVec = m_transMatTranspose * m_filtVec; // now p(x_t |y_{1:t-1})
364  m_filtVec = m_filtVec.cwiseProduct( condDensVec ); // now p(y_t,x_t|y_{1:t-1})
365  m_lastCondLike = m_filtVec.sum();
366  m_filtVec /= m_lastCondLike; // now p(x_t|y_{1:t})
367  }
