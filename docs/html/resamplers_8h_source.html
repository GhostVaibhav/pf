120  // these log weights may be very negative. If that's the case, exponentiating them may cause underflow
121  // so we use the "log-exp-sum" trick
122  // actually not quite...we just shift the log-weights because after they're exponentiated
123  // they have the same normalized probabilities
124 
125  // Create the distribution with exponentiated log-weights
126  arrayFloat w;
127  float_t m = *std::max_element(oldLogUnNormWts.begin(), oldLogUnNormWts.end());
128  std::transform(oldLogUnNormWts.begin(), oldLogUnNormWts.end(), w.begin(),
129  [&m](float_t& d) -> float_t { return std::exp( d - m ); } );
130  std::discrete_distribution<> idxSampler(w.begin(), w.end());
131
