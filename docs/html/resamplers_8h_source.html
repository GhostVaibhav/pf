117 void mn_resampler<nparts, dimx>::resampLogWts(arrayVec &oldParts, arrayDouble &oldLogUnNormWts)
118 {
119  // these log weights may be very negative. If that's the case, exponentiating them may cause underflow
120  // so we use the "log-exp-sum" trick
121  // actually not quite...we just shift the log-weights because after they're exponentiated
122  // they have the same normalized probabilities
123 
124  // Create the distribution with exponentiated log-weights
125  arrayDouble w;
126  double m = *std::max_element(oldLogUnNormWts.begin(), oldLogUnNormWts.end());
127  std::transform(oldLogUnNormWts.begin(), oldLogUnNormWts.end(), w.begin(),
128  [&m](double& d) -> double { return std::exp( d - m ); } );
129  std::discrete_distribution<> idxSampler(w.begin(), w.end());
130 
131  // create temporary particle vector and weight vector
132  arrayVec tmpPartics = oldParts;
133 
134  // sample from the original parts and store in tmpParts
135  unsigned int whichPart;
136  for(size_t part = 0; part < nparts; ++part)
137  {
138  whichPart = idxSampler(m_gen);
139  tmpPartics[part] = oldParts[whichPart];
140  }
141 
142  //overwrite olds with news
143  oldParts = std::move(tmpPartics);
144  std::fill(oldLogUnNormWts.begin(), oldLogUnNormWts.end(), 0.0); // change back
145 }
