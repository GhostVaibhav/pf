200 template<std::size_t dim>
201 double evalMultivNorm(const Eigen::Matrix<double,dim,1> &x,
202  const Eigen::Matrix<double,dim,1> &meanVec,
203  const Eigen::Matrix<double,dim,dim> &covMat,
204  bool log = false)
205 {
206  using Mat = Eigen::Matrix<double,dim,dim>;
207 
208  // from Eigen: Remember that Cholesky decompositions are not rank-revealing.
210  // use LDLT instead for the semidefinite case. Also, do not use a Cholesky
211  // decomposition to determine whether a system of equations has a solution.
212  Eigen::LLT<Mat> lltM(covMat);
213  double quadform = (lltM.matrixL().solve(x-meanVec)).squaredNorm();
214  if (log){
215 
216  // calculate log-determinant using cholesky decomposition too
217  double ld (0.0);
218  Mat L = lltM.matrixL(); // the lower diagonal L such that M = LL^T
219 
220  // add up log of diagnols of Cholesky L
221  for(size_t i = 0; i < dim; ++i){
222  ld += std::log(L(i,i));
223  }
224  ld *= 2; // covMat = LL^T
225 
226  return -.5*log_two_pi * dim - .5*ld - .5*quadform;
227 
228 
229  }else{ // not the log density
230  double normConst = std::pow(inv_sqrt_2pi, dim) / lltM.matrixL().determinant();
231  return normConst * std::exp(-.5* quadform);
232  }
233 }

248 double evalMultivNormWBDA(const Eigen::Matrix<double,bigd,1> &x,
249  const Eigen::Matrix<double,bigd,1> &meanVec,

253  bool log = false)
254 {
255 
256  using bigmat = Eigen::Matrix<double,bigd,bigd>;
257  using smallmat = Eigen::Matrix<double,smalld,smalld>;
258 
259  bigmat Ainv = A.asDiagonal().inverse();
260  smallmat Cinv = C.inverse();
261  smallmat I = Cinv + U.transpose()*Ainv*U;
262  bigmat SigInv = Ainv - Ainv * U * I.ldlt().solve(U.transpose() * Ainv);
263  Eigen::LLT<bigmat> lltSigInv(SigInv);
264  bigmat L = lltSigInv.matrixL(); // LL' = Sig^{-1}
265  double quadform = (L * (x-meanVec)).squaredNorm();
