203 template<std::size_t dim>
204 double evalMultivNorm(const Eigen::Matrix<double,dim,1> &x,
205  const Eigen::Matrix<double,dim,1> &meanVec,
206  const Eigen::Matrix<double,dim,dim> &covMat,
207  bool log = false)
208 {
209  using Mat = Eigen::Matrix<double,dim,dim>;
210 
211  // from Eigen: Remember that Cholesky decompositions are not rank-revealing.
213  // use LDLT instead for the semidefinite case. Also, do not use a Cholesky
214  // decomposition to determine whether a system of equations has a solution.
215  Eigen::LLT<Mat> lltM(covMat);
216  double quadform = (lltM.matrixL().solve(x-meanVec)).squaredNorm();
217  if (log){
218 
219  // calculate log-determinant using cholesky decomposition too
220  double ld (0.0);
221  Mat L = lltM.matrixL(); // the lower diagonal L such that M = LL^T
222 
223  // add up log of diagnols of Cholesky L
224  for(size_t i = 0; i < dim; ++i){
225  ld += std::log(L(i,i));
226  }
227  ld *= 2; // covMat = LL^T
228 
229  return -.5*log_two_pi * dim - .5*ld - .5*quadform;
230 
231 
232  }else{ // not the log density
233  double normConst = std::pow(inv_sqrt_2pi, dim) / lltM.matrixL().determinant();
234  return normConst * std::exp(-.5* quadform);
235  }
236 }

250 template<std::size_t bigd, std::size_t smalld>
251 double evalMultivNormWBDA(const Eigen::Matrix<double,bigd,1> &x,
252  const Eigen::Matrix<double,bigd,1> &meanVec,
253  const Eigen::Matrix<double,bigd,1> &A,
254  const Eigen::Matrix<double,bigd,smalld> &U,
255  const Eigen::Matrix<double,smalld,smalld> &C,
256  bool log = false)
257 {
258 
259  using bigmat = Eigen::Matrix<double,bigd,bigd>;
260  using smallmat = Eigen::Matrix<double,smalld,smalld>;
261 
262  bigmat Ainv = A.asDiagonal().inverse();
263  smallmat Cinv = C.inverse();
264  smallmat I = Cinv + U.transpose()*Ainv*U;
265  bigmat SigInv = Ainv - Ainv * U * I.ldlt().solve(U.transpose() * Ainv);
266  Eigen::LLT<bigmat> lltSigInv(SigInv);
267  bigmat L = lltSigInv.matrixL(); // LL' = Sig^{-1}
268  double quadform = (L * (x-meanVec)).squaredNorm();
269  if (log){
270 
271  // calculate log-determinant using cholesky decomposition (assumes symmetric and positive definite)
272  double halfld (0.0);
273  // add up log of diagnols of Cholesky L
274  for(size_t i = 0; i < bigd; ++i){
275  halfld += std::log(L(i,i));
276  }
277 
278  return -.5*log_two_pi * bigd + halfld - .5*quadform;
279 
280 
281  }else{ // not the log density
282  double normConst = std::pow(inv_sqrt_2pi, bigd) * L.determinant();
283  return normConst * std::exp(-.5* quadform);
